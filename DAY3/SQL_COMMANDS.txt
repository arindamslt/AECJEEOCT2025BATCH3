DATA RETRIEVAL
==========================================================================
ALL DATA
=======================================

========================================
SELECT ENAME,JOB,SAL FROM EMP;
RESTRICTION OF DATA
==========================================================
IT IS DONE BY WHERE CLAUSE
SELECT * FROM EMP WHERE ENAME='SUMAN';
SELECT * FROM EMP WHERE JOB='CLERK';
==========================================================
IN OPERATOR
=====================================================
SELECT * FROM EMP WHERE JOB IN('CLERK','ANALYST');
SELECT * FROM EMP WHERE JOB NOT IN('CLERK','ANALYST');
=======================================================
AND OPERATOR
=================================================
SELECT * FROM EMP WHERE JOB='CLERK' AND SAL > 25000;
==================================================
OR OPERATOR
SELECT * FROM EMP WHERE JOB='CLERK' OR SAL > 25000;
=======================================================
LIKE OPERATOR
===========================================================
SELECT * FROM EMP WHERE ENAME LIKE 'S%';
SELECT * FROM EMP WHERE ENAME LIKE '_Y%';
SELECT * FROM EMP WHERE ENAME LIKE '%I';
SELECT * FROM EMP WHERE ENAME LIKE '%A%';
